-- Insight 1 - top 5 days with maximum accidents
-- requirement: top 5 days, the boroughs where they occurred and the vehicle type (code 1)
CREATE OR REPLACE TABLE INSIGHT_TOP_5_DAYS AS 
WITH Top5Dates AS (
    SELECT ACCIDENT_DATE, COUNT(*) AS DAILY_TOTAL
    FROM NYPD_COLLISIONS_FINAL
    GROUP BY 1
    ORDER BY DAILY_TOTAL DESC
    LIMIT 5
)
SELECT 
    f.ACCIDENT_DATE,
    f.CLEANED_BOROUGH,
    f."VEHICLE TYPE CODE 1",
    COUNT(*) AS ACCIDENT_COUNT
FROM NYPD_COLLISIONS_FINAL f
JOIN Top5Dates t
ON f.ACCIDENT_DATE = t.ACCIDENT_DATE
GROUP BY 1, 2, 3
ORDER BY f.ACCIDENT_DATE ASC, ACCIDENT_COUNT DESC;

SELECT * FROM INSIGHT_TOP_5_DAYS;



-- Insight 2 - Maximum accidents borough & its top 5 days
--Requirement: which borough recorded the max accidents, and the top 5 days for that specific Borough.
CREATE OR REPLACE TABLE INSIGHT_BOROUGH_PEAKS AS
WITH TopBorough AS (
    SELECT CLEANED_BOROUGH, COUNT(*) as TOTAL_ACCIDENTS
    FROM NYPD_COLLISIONS_FINAL
    GROUP BY 1
    ORDER BY TOTAL_ACCIDENTS DESC
    LIMIT 1
)
SELECT 
    f.CLEANED_BOROUGH, 
    f.ACCIDENT_DATE, 
    COUNT(*) as ACCIDENT_COUNT
FROM NYPD_COLLISIONS_FINAL f
WHERE f.CLEANED_BOROUGH = (SELECT CLEANED_BOROUGH FROM TopBorough)
GROUP BY 1, 2
ORDER BY ACCIDENT_COUNT DESC
LIMIT 5;

SELECT * FROM INSIGHT_BOROUGH_PEAKS;



-- Insight 3: top 5 vehicle types
-- Requirement - top 5 vehicle types involved in maximum accidents.
CREATE OR REPLACE TABLE INSIGHT_TOP_VEHICLES AS
SELECT "VEHICLE TYPE CODE 1", COUNT(*) as ACCIDENT_COUNT
FROM NYPD_COLLISIONS_FINAL
WHERE "VEHICLE TYPE CODE 1" IS NOT NULL
GROUP BY 1
ORDER BY ACCIDENT_COUNT DESC
LIMIT 5;

SELECT * FROM INSIGHT_TOP_VEHICLES;
