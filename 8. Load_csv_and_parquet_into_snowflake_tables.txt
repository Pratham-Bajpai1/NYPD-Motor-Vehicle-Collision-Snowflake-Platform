-- Requirement - insert the generated csv/parquet files into snowflake tables (fix parquet part not working)

-- Table for insight 1 from s3 
-- csv
CREATE OR REPLACE TABLE LOADED_INSIGHT_TOP_5_DAYS_CSV (
    ACCIDENT_DATE DATE,
    CLEANED_BOROUGH STRING,
    VEHICLE_TYPE STRING,
    ACCIDENT_COUNT INT
);

COPY INTO LOADED_INSIGHT_TOP_5_DAYS_CSV
FROM @nypd_external_stage/output/insight_top_5_days/
FILE_FORMAT = (TYPE = 'CSV' FIELD_OPTIONALLY_ENCLOSED_BY = '"' SKIP_HEADER = 1);

SELECT * FROM LOADED_INSIGHT_TOP_5_DAYS_CSV;

-- parquet
CREATE OR REPLACE TABLE LOADED_INSIGHT_TOP_5_DAYS_PARQUET AS
SELECT * FROM LOADED_INSIGHT_TOP_5_DAYS_CSV WHERE 1=0;

COPY INTO LOADED_INSIGHT_TOP_5_DAYS_PARQUET
FROM @nypd_external_stage/output/insight_top_5_days_parquet/
FILE_FORMAT = (TYPE = PARQUET);

SELECT * FROM LOADED_INSIGHT_TOP_5_DAYS_PARQUET;
