CREATE OR REPLACE TABLE NYPD_COLLISIONS_RAW (
    "ACCIDENT_DATE" DATE,
    "ACCIDENT_TIME" TIME,
    "BOROUGH" STRING,
    "ZIP CODE" STRING,
    "LATITUDE" FLOAT,
    "LONGITUDE" FLOAT,
    "LOCATION" STRING,
    "ON STREET NAME" STRING,
    "CROSS STREET NAME" STRING,
    "OFF STREET NAME" STRING,
    "NUMBER OF PERSONS INJURED" INT,
    "NUMBER OF PERSONS KILLED" INT,
    "NUMBER OF PEDESTRIANS INJURED" INT,
    "NUMBER OF PEDESTRIANS KILLED" INT,
    "NUMBER OF CYCLIST INJURED" INT,
    "NUMBER OF CYCLIST KILLED" INT,
    "NUMBER OF MOTORIST INJURED" INT,
    "NUMBER OF MOTORIST KILLED" INT,
    "CONTRIBUTING FACTOR VEHICLE 1" STRING,
    "CONTRIBUTING FACTOR VEHICLE 2" STRING,
    "CONTRIBUTING FACTOR VEHICLE 3" STRING,
    "CONTRIBUTING FACTOR VEHICLE 4" STRING,
    "CONTRIBUTING FACTOR VEHICLE 5" STRING,
    "COLLISION_ID" INT PRIMARY KEY,
    "VEHICLE TYPE CODE 1" STRING,
    "VEHICLE TYPE CODE 2" STRING,
    "VEHICLE TYPE CODE 3" STRING,
    "VEHICLE TYPE CODE 4" STRING,
    "VEHICLE TYPE CODE 5" STRING
);



COPY INTO NYPD_COLLISIONS_RAW
FROM @nypd_external_stage/nypd-motor-vehicle-collisions.csv
FILE_FORMAT = (FORMAT_NAME = nypd_csv_format)
ON_ERROR = 'CONTINUE';


-- Sub-task a.1 - Total number of rows
SELECT COUNT(*) AS TOTAL_ROWS FROM NYPD_COLLISIONS_RAW;

-- Save into table
CREATE OR REPLACE TABLE RAW_COUNT_TABLE_NYPD_COLLISION AS
SELECT COUNT(*) AS TOTAL_ROWS
FROM NYPD_COLLISIONS_RAW;

-- Save as csv(export to s3) -- Not executed update iam policy
COPY INTO @nypd_external_stage/row_count_nypd_collision_table/
FROM (
    SELECT COUNT(*) AS TOTAL_ROWS FROM NYPD_COLLISIONS_RAW
)
FILE_FORMAT = (TYPE = CSV FIELD_OPTIONALLY_ENCLOSED_BY = '"')
OVERWRITE = TRUE;


-- Sub-task a.2 - Rows missing Borough OR Zip Code
SELECT COUNT(*) AS MISSING_BOROUGH_OR_ZIP
FROM NYPD_COLLISIONS_RAW
WHERE BOROUGH IS NULL OR 'ZIP CODE' IS NULL;
