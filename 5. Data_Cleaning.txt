-- We ignore rows where Lat/Long are missing as per project requirements
-- only unique values for Latitude and Longitude wiht no decimals

CREATE OR REPLACE TABLE LAT_LONG_MAPPING AS
WITH FrequencyMapping AS (
    SELECT 
        TRUNC(LATITUDE) AS LATITUDE_KEY, 
        TRUNC(LONGITUDE) AS LONGITUDE_KEY, 
        BOROUGH, 
        "ZIP CODE",
        COUNT(*) as occurrence_count
    FROM NYPD_COLLISIONS_RAW
    WHERE LATITUDE IS NOT NULL AND LATITUDE != 0  
      AND LONGITUDE IS NOT NULL AND LONGITUDE != 0
      AND BOROUGH IS NOT NULL 
      AND "ZIP CODE" IS NOT NULL
    GROUP BY 1, 2, 3, 4
)
SELECT LATITUDE_KEY, LONGITUDE_KEY, BOROUGH, "ZIP CODE"
FROM FrequencyMapping
-- this line picks the most frequent borough or zip for each coordinate block
QUALIFY ROW_NUMBER() OVER (PARTITION BY LATITUDE_KEY, LONGITUDE_KEY ORDER BY occurrence_count DESC) = 1;

SELECT * FROM LAT_LONG_MAPPING;


-- performing data cleansing 
-- creating cleaned table, joining our raw data with the lat_long_mapping to fill the Borough/ZIP 
CREATE OR REPLACE TABLE NYPD_COLLISIONS_CLEANED AS
SELECT 
    r.*,
    COALESCE(r.BOROUGH, m.BOROUGH) AS CLEANED_BOROUGH,
    COALESCE(r."ZIP CODE", m."ZIP CODE") AS CLEANED_ZIP_CODE
FROM NYPD_COLLISIONS_RAW r
LEFT JOIN LAT_LONG_MAPPING m
    ON TRUNC(r.LATITUDE) = m.LATITUDE_KEY
    AND TRUNC(r.LONGITUDE) = m.LONGITUDE_KEY
WHERE r.LATITUDE IS NOT NULL AND r.LATITUDE != 0 
  AND r.LONGITUDE IS NOT NULL AND r.LONGITUDE != 0;

SELECT * FROM NYPD_COLLISIONS_CLEANED;


-- removing remaining missing rows, 
CREATE OR REPLACE TABLE NYPD_COLLISIONS_FINAL AS
SELECT * FROM NYPD_COLLISIONS_CLEANED
WHERE CLEANED_BOROUGH IS NOT NULL 
  AND CLEANED_ZIP_CODE IS NOT NULL;

SELECT * FROM NYPD_COLLISIONS_FINAL;  


-- validation Queries (Sub-task a.3)
-- Remaining Missing Rows
SELECT COUNT(*) AS STILL_MISSING_COUNT
FROM NYPD_COLLISIONS_FINAL
WHERE CLEANED_BOROUGH IS NULL
   OR CLEANED_ZIP_CODE IS NULL;

-- Final Clean Row Count    
SELECT COUNT(*) FROM NYPD_COLLISIONS_FINAL;

-- Rows Corrected or Filled
SELECT 
    COUNT(*) AS FILLED_ROWS
FROM NYPD_COLLISIONS_FINAL
WHERE (BOROUGH IS NULL OR "ZIP CODE" IS NULL)
   AND (CLEANED_BOROUGH IS NOT NULL OR CLEANED_ZIP_CODE IS NOT NULL);
